# Railway Environment Variables Configuration
# Copy these variables to your Railway project dashboard under "Variables" tab

# === Laravel Application Configuration ===
APP_NAME=Housync
APP_ENV=production
APP_KEY=base64:YOUR_GENERATED_APP_KEY_HERE
APP_DEBUG=false
APP_URL=https://your-app-name.up.railway.app

# === Database Configuration ===
# Railway will auto-populate these when you add MySQL service
DB_CONNECTION=mysql
DB_HOST=${{ MYSQL_HOST }}
DB_PORT=${{ MYSQL_PORT }}
DB_DATABASE=${{ MYSQL_DATABASE }}
DB_USERNAME=${{ MYSQL_USER }}
DB_PASSWORD=${{ MYSQL_PASSWORD }}

# Alternative: Use DATABASE_URL (Railway provides this automatically)
# DATABASE_URL=${{ DATABASE_URL }}

# === Session & Cache Configuration ===
SESSION_DRIVER=database
SESSION_LIFETIME=120
SESSION_ENCRYPT=false
SESSION_PATH=/
SESSION_DOMAIN=null
CACHE_STORE=database
CACHE_PREFIX=
QUEUE_CONNECTION=database

# === Security Configuration ===
SESSION_SECURE_COOKIE=true
SESSION_SAME_SITE=lax
TRUSTED_PROXIES=*

# === Mail Configuration (Optional) ===
MAIL_MAILER=log
MAIL_HOST=127.0.0.1
MAIL_PORT=2525
MAIL_USERNAME=null
MAIL_PASSWORD=null
MAIL_ENCRYPTION=null
MAIL_FROM_ADDRESS=noreply@your-domain.com
MAIL_FROM_NAME="${APP_NAME}"

# === File Storage Configuration ===
FILESYSTEM_DISK=public

# For production file storage (recommended):
# FILESYSTEM_DISK=s3
# AWS_ACCESS_KEY_ID=your_access_key
# AWS_SECRET_ACCESS_KEY=your_secret_key
# AWS_DEFAULT_REGION=us-east-1
# AWS_BUCKET=your-bucket-name

# === RFID Security System Configuration ===
RFID_READER_LOCATION=main_entrance
RFID_ACCESS_LOG_RETENTION_DAYS=365
RFID_DEFAULT_CARD_EXPIRY_DAYS=365

# === Logging Configuration ===
LOG_CHANNEL=stack
LOG_STACK=single
LOG_DEPRECATIONS_CHANNEL=null
LOG_LEVEL=info

# === Broadcasting Configuration (if needed) ===
BROADCAST_CONNECTION=log

# === Additional Laravel Configuration ===
APP_LOCALE=en
APP_FALLBACK_LOCALE=en
APP_FAKER_LOCALE=en_US
APP_MAINTENANCE_DRIVER=file
APP_TIMEZONE=UTC

# === Instructions ===
# 1. Generate APP_KEY by running locally: php artisan key:generate --show
# 2. Replace "your-app-name" with your actual Railway app name
# 3. Add MySQL service in Railway dashboard
# 4. Railway will automatically populate MySQL variables
# 5. Set APP_DEBUG=true for initial testing, then false for production
